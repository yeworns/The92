      function logVisit() {

        const date = document.getElementById("visit-date").value;

        const notes = document.getElementById("notes").value;

        const ticketFile = document.getElementById("ticket").files[0];


        if (!date) return alert("Please select a date.");


        const reader = new FileReader();

        reader.onloadend = () => {

          fetchWeatherAndScore(date, (weather, score) => {

            visited.push({

              ...selectedStadium,

              date,

              notes,

              ticket: reader.result,

              weather,

              score

            });

            selectedStadium = null;

            renderForm();

            renderVisited();

          });

        };


        if (ticketFile) {

          reader.readAsDataURL(ticketFile);

        } else {

          fetchWeatherAndScore(date, (weather, score) => {

            visited.push({

              ...selectedStadium,

              date,

              notes,

              ticket: null,

              weather,

              score

            });

            selectedStadium = null;

            renderForm();

            renderVisited();

          });

        }

      }


      function fetchWeatherAndScore(date, callback) {

        const [lat, lon] = selectedStadium.coords;


        let weatherData = 'Loading...';

        let scoreData = 'Loading...';


        // Weather API

        fetch(`https://api.openweathermap.org/data/2.5/weather?lat=${lat}&lon=${lon}&appid=${WEATHER_API_KEY}&units=metric`)

          .then(res => res.json())

          .then(data => {

            weatherData = `${data.weather[0].main}, ${data.main.temp}°C`;

          })

          .catch(() => {

            weatherData = 'Weather not found';

          })

          .finally(() => {

            fetch(`https://v3.football.api-sports.io/fixtures?date=${date}`, {

              headers: { 'x-apisports-key': FOOTBALL_API_KEY }

            })

              .then(res => res.json())

              .then(data => {

                const match = data.response.find(m => m.teams.home.name === selectedStadium.club || m.teams.away.name === selectedStadium.club);

                if (match) {

                  scoreData = `${match.teams.home.name} ${match.goals.home} - ${match.goals.away} ${match.teams.away.name}`;

                } else {

                  scoreData = 'Score not found';

                }

              })

              .catch(() => {

                scoreData = 'Score not found';

              })

              .finally(() => callback(weatherData, scoreData));

          });

      }

    function renderTopStats() {

  if (!visited.length) {

    document.getElementById('top-stats').innerHTML = '';

    return;

  }


  const mostRecent = visited[visited.length - 1];

  const coldest = visited.reduce((min, v) => {

    const temp = parseFloat((v.weather || '').match(/([-\d.]+)°C/)?.[1]);

    return !isNaN(temp) && temp < min.temp ? { stadium: v.name, temp } : min;

  }, { stadium: '', temp: Infinity });


  const hottest = visited.reduce((max, v) => {

    const temp = parseFloat((v.weather || '').match(/([-\d.]+)°C/)?.[1]);

    return !isNaN(temp) && temp > max.temp ? { stadium: v.name, temp } : max;

  }, { stadium: '', temp: -Infinity });


  const totalNotes = visited.reduce((sum, v) => sum + (v.notes?.length || 0), 0);

  const averageNoteLength = (totalNotes / visited.length).toFixed(1);


  const ticketsUploaded = visited.filter(v => v.ticket).length;


  document.getElementById('top-stats').innerHTML = `

    <strong>Top Stats:</strong><br>

    🔁 Last Visit: ${mostRecent.name} (${mostRecent.date})<br>

    🌡️ Coldest Match: ${coldest.stadium} (${coldest.temp}°C)<br>

    🔥 Hottest Match: ${hottest.stadium} (${hottest.temp}°C)<br>

    📝 Avg. Notes Length: ${averageNoteLength} chars<br>

    🎟️ Ticket Uploads: ${ticketsUploaded} of ${visited.length}

  `;

}
